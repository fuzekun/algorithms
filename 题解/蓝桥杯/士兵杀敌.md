# 士兵杀敌

> 持续更新[蓝桥杯算法训练题解](https://blog.csdn.net/fuzekun/article/details/127298556?spm=1001.2014.3001.5502)，有兴趣可以关注一波呀。



直接就是树状数组的最基本题目，没什么好说的，错了一次，**读题不认真**，没看见多组测试数据。

另外学到了一个东西，scanf("%s", input);是以**空格和回车符**为终止的，**空格也会终止**

还有就是Java中输入以EOF结尾`while (sc.hasNext())`，当然输入嘛，也有坑：

Java中的nextInt不会把换行符吃掉，如果直接sc.nextLine()会得到一个换行符，所以应该先sc.nextLine()吃掉换行符之后再读入下一行。

当然最简单的方式是**只按照sc.nextLine()读入**，然后使用空格分隔得到每一行的输入的数字和其他东西。这样就不用考虑输入的换行符的问题了。

- c++代码



```cpp
/*

    士兵杀敌

    直接是裸的树状数组
*/

#include <bits/stdc++.h>

using namespace std;

const int maxn = 1e5 +5;

int c[maxn];

int lowbit(int x) {
    return x & -x;
}
int query(int x) {
    int res = 0;
    for (int i = x; i; i -= lowbit(i))
        res += c[i];
    return res;
}
int add(int x, int d) {
    for (int i = x; i < maxn; i += lowbit(i))
        c[i] += d;
}

int main() {


    int n, m;
    while (scanf("%d%d", &n, &m) != EOF) {

        memset(c, 0, sizeof c);
        for (int i = 0; i < n; i++) {
            int a;
            scanf("%d", &a);
            add(i + 1, a);
        }
        char in[25];
        for (int i = 0; i < m; i++) {
            scanf("%s", in);
            //cout << in << endl;
            if (in[0] == 'Q') {
                int x = 0, y = 0;
                scanf("%s", in);
                for (int j = 0; in[j]; j++) x = x * 10 + (in[j] - '0');
                scanf("%s", in);
                for (int k = 0; in[k]; k++) y = y * 10 + (in[k] - '0');
                printf("%d\n", query(y) - query(x - 1));
            } else {
                int x = 0, y = 0;
                scanf("%s", in);
                for (int j = 0; in[j]; j++) x = x * 10 + (in[j] - '0');
                scanf("%s", in);
                for (int k = 0; in[k]; k++) y = y * 10 + (in[k] - '0');
                add(x, y);
            }
        }
    }
    return 0;
}
```

- java代码

```java
package lanqiao;

import java.util.Arrays;
import java.util.Scanner;

/**
 * @author: Zekun Fu
 * @date: 2022/10/14 10:41
 * @Description: 士兵杀敌
 */
public class ShiBingShaDi {
    private static final int maxn = (int)1e5 +5;

    private static int[] c = new int[maxn];

    private static int lowbit(int x) {
        return x & -x;
    }
    private static int query(int x) {
        int res = 0;
        for (int i = x; i != 0; i -= lowbit(i))
            res += c[i];
        return res;
    }
    private static void add(int x, int d) {
        for (int i = x; i < maxn; i += lowbit(i))
            c[i] += d;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        while (sc.hasNext()) {
            Arrays.fill(c, 0);
            int n = sc.nextInt();
            int m = sc.nextInt();
            for (int i = 0; i < n; i++) {
                int a = sc.nextInt();
                add(i + 1, a);
            }
            sc.nextLine();									// 吃掉nextInt后面的换行符
            for (int i = 0; i < m; i++) {
                String s = sc.nextLine();
                String[] sp = s.split(" ");
                int x = Integer.valueOf(sp[1]);
                int y = Integer.valueOf(sp[2]);
                if (sp[0].compareTo("QUERY") == 0) {
                    System.out.println(query(y) - query(x - 1));
                } else add(x, y);
            }
        }
    }
}
```
